use std::rc::Rc;
use std::time::Duration;

use core::socket::{
    SocketAddrLease,
    TcpDefaultSeqNum,
};
use core::time::Env as TimeEnv;

/// An execution context for TCP streams and listeners.
#[derive(Clone, Debug)]
pub struct TcpContext {
    /// The local address a stream or listener is bound to.
    pub binding: Rc<SocketAddrLease>,
    /// An environment providing the current time.
    pub time_env: Rc<TimeEnv>,
    /// An allocator for assigning initial sequence numbers to new connections.
    pub def_seq_num: Rc<TcpDefaultSeqNum>,
    /// The default RTO, used before enough segment RTT samples are gathered to
    /// estimate the true RTO.
    pub retransmit_timeout: Duration,
    /// The timeout for a SYN_RECV stream that fails to transition to an
    /// ESTABLISHED state before being dropped by a listener.
    pub syn_recv_timeout: Duration,
    /// The number of streams in the SYN_RECV or ESTABLISHED state the listener
    /// can maintain that have not been dequeued with an accept(...) call
    /// before ignoring new connection requests.
    pub connection_queue_len: usize,
    /// The maximum receive window size  of a stream
    pub send_queue_len: usize,
}
